import string
from nltk.corpus import stopwords
from nltk.tokenize import word_tokenize

# Sample text
text = "Hello!!! This is an Example, with PUNCTUATION and StopWords."

# Step 1: Lowercase
text = text.lower()

# Step 2: Remove punctuation
text = text.translate(str.maketrans('', '', string.punctuation))

# Step 3: Tokenize
tokens = word_tokenize(text)

# Step 4: Remove stopwords
stop_words = set(stopwords.words('english'))
cleaned_tokens = [word for word in tokens if word not in stop_words]

print("Original Text:", "Hello!!! This is an Example, with PUNCTUATION and StopWords.")
print("Cleaned Tokens:", cleaned_tokens)
Code Description:

Converts text to lowercase for consistency.

Removes punctuation using Python’s string library.

Tokenizes text into words.

Removes stopwords (like “is,” “an,” “with”) using NLTK.

Outputs cleaned tokens for further NLP tasks.
