# Smart Input Program

name = input("Enter your name: ")
age = int(input("Enter your age: "))
hobby = input("Enter your hobby: ")

# Categorize age
if age < 13:
    category = "Child"
elif age < 20:
    category = "Teenager"
elif age < 60:
    category = "Adult"
else:
    category = "Senior"

# Personalized message
print(f"Hello {name}! You are {age} years old, categorized as a {category}.")
print(f"It's great that you enjoy {hobby}!")
