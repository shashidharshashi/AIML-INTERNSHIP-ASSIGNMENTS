import hashlib
import os

def hash_password(password, salt):
    return hashlib.pbkdf2_hmac('sha256', password.encode(), salt, 100000)

def create_user(username, password):
    salt = os.urandom(16)
    hashed = hash_password(password, salt)
    return {"username": username, "salt": salt, "hash": hashed}

def authenticate(user, attempt):
    attempt_hash = hash_password(attempt, user["salt"])
    return attempt_hash == user["hash"]

# Example run
if __name__ == "__main__":
    uname = input("Enter username: ")
    pwd = input("Enter password: ")
    user = create_user(uname, pwd)

    print("\nLogin now")
    attempt = input("Password: ")
    if authenticate(user, attempt):
        print("Login successful")
    else:
        print("Login failed")
