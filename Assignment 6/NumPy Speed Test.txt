#Compare Python lists vs NumPy arrays with 1M numbers, measure execution time, write 3 observations.
import time
import numpy as np

# Python list test
py_list = list(range(1_000_000))
start = time.time()
py_list_squared = [x**2 for x in py_list]
end = time.time()
print("Python list time:", end - start, "seconds")

# NumPy array test
np_array = np.arange(1_000_000)
start = time.time()
np_array_squared = np_array ** 2
end = time.time()
print("NumPy array time:", end - start, "seconds")
